# Example scene configuration for NeoHookean simulation
# Lines starting with # are comments

simulation {
    time_step 0.01
    friction_coef 0.11
    stiffness_coef 4e4
    tol 0.01
    dhat 0.5
    kappa 1e8
    num_steps 100

    # Ground plane (point and normal)
    ground_x 0.0
    ground_y -1.0
    ground_z 0.0
    ground_nx 0.0
    ground_ny 1.0
    ground_nz 0.0
}

# Generate a 3x3x2 = 18 instances of spheres in a 3D grid
mesh {
    filepath "input/el_topo_sphere_1280.obj"

    # Material properties (shared by all instances)
    density 1000.0
    young_mod 1e5
    poisson_ratio 0.4
    bending_stiff 1e8

    # 3D grid layout: 3 in X, 3 in Y, 2 in Z = 18 total instances
    num_instances_x 5
    num_instances_y 2
    num_instances_z 5

    # Spacing between instances
    spacing_x 2.5
    spacing_y 2.5
    spacing_z 2.5

    # Grid starting position (lower-left-front corner)
    offset_x -2.5
    offset_y 2.0
    offset_z -2.5

    # Scale applied to each instance
    scale 0.5

    # Initial velocity (shared by all instances)
    velocity_x 0.0
    velocity_y 0.0
    velocity_z 0.0

    # Constraints
    fixed_x false
    fixed_y false
    fixed_z false
}
